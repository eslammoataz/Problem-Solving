//O(sqrt(n))
bool isPrime(int n) {
    if (n == 1 || n == 0) return false;
    for (int i = 2; i <= n / i; i += 1 + (i & 1))
        if (n % i == 0) return false;
 
    return true;
}

bool isPrime(int n) {
    for (int i = 2; i <= n / i; i += 1 + (i & 1) )
        if (n%i == 0) return 0;
 
        return n > 1;
}
